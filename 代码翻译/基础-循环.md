## C / C++

```C
//for循环
int i;
for (i=0;i<10;i++) {
    //退出循环
    break;
    //跳过本次循环
    continue;
}

//while循环
int i = 0;
while(i<10) {
    i++;
}

//do-while循环
int i = 0;
do{
    i++;
} while(i<10);

```

## Lua

```Lua
-- for循环，第三个参数默认为1，即每次+1
-- for i=10, 1, -1 do
for i=0, 10 do
    -- 退出循环
    break
    -- 跳过循环使用goto
    goto continue
    ::continue::
end

-- for 泛型循环
a = {"one", "two", "three"}
for i, v in ipairs(a) do
    --print(i, v)
    -- 退出循环
    break
    -- 跳过循环使用goto
    goto continue
    ::continue::
end 

-- while循环
i = 0
while(i < 10)
do
    -- 退出循环
    break
    -- 跳过循环使用goto
    goto continue
    ::continue::
    i = i + 1
end

-- until循环
i = 0
repeat
    -- 退出循环
    break
    -- 跳过循环使用goto
    goto continue
    ::continue::
    i = i + 1
until(i < 10)

```

## Php


```php
//for循环
for($i=0;$i<10;$i++) {
    //退出循环
    break;
    //跳过本次循环
    continue;
}

//foreach循环
//foreach($array as $value) {
foreach($array as $key => $value) {
    //退出循环
    break;
    //跳过本次循环
    continue;
}

//while循环
while($i<10) {
    //退出循环
    break;
    //跳过本次循环
    continue;
    $i++;
}

//do-while循环
int i = 0;
do {
    //退出循环
    break;
    //跳过本次循环
    continue;
    $i++;
} while($i<10);
```
